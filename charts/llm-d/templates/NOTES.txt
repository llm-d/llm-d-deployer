    name: basic-gpu-with-nixl-and-redis-lookup-preset

    modelName: <model_name>
  modelArtifacts:
    uri: pvc://<pvc_name>/<path_to_model>

    replicas: 1
    containers:
    - name: "vllm"

      - <model_name>
